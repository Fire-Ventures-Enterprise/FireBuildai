// SMS Service using Twilio
class SMSNotificationService {
  async sendETANotification(customer: Customer, contractor: User, eta: number) {
    const message = `Hi ${customer.name}! Your contractor ${contractor.name} 
    is on the way to ${customer.address}. ETA: ${eta} minutes. 
    Track live: ${this.generateTrackingLink(contractor.id)}`;
    
    await this.twilioClient.messages.create({
      body: message,
      from: process.env.TWILIO_PHONE_NUMBER,
      to: customer.phone
    });
  }

  async sendArrivalNotification(customer: Customer, contractor: User) {
    const message = `${contractor.name} has arrived at your location 
    and is ready to start work. Job #${job.id}`;
    
    await this.sendSMS(customer.phone, message);
  }

    return `https://firebuild.ai/track/${contractorId}`;
  }
}
